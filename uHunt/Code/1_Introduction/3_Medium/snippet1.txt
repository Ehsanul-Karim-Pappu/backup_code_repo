#include <stdio.h>

char *fgets_wrapper(char *buffer, size_t buflen, FILE *fp)
{
    if (fgets(buffer, buflen, fp) != 0)
    {
        buffer[strcspn(buffer, "\n")] = '\0'; //strcspn search the string(buffer) for another string("\n") and returns the index where is found..
        return buffer;
    }
    return 0;
}
